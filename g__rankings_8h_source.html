00027 /*
00028 ==============================================================================
00029 
00030 Key digits:
00031         10^9: report type
00032                         1 = normal
00033                         2 = developer-only
00034         10^8: stat type
00035                         0 = match stat
00036                         1 = single player stat
00037                         2 = duel stat
00038         10^7: data type
00039                         0 = string
00040                         1 = uint32
00041         10^6: calculation
00042                         0 = use raw value
00043                         1 = add to total
00044                         2 = average
00045                         3 = max
00046                         4 = min
00047         10^5
00048         10^4: category
00049                         00 = general
00050                         01 = session
00051                         02 = weapon
00052                         03 = ammo
00053                         04 = health
00054                         05 = armor
00055                         06 = powerup
00056                         07 = holdable
00057                         08 = hazard
00058                         09 = reward
00059                         10 = teammate
00060                         11 = ctf
00061         10^3:
00062         10^2: sub-category
00063         10^1:
00064         10^0: ordinal
00065  
00066 ==============================================================================
00067 */
