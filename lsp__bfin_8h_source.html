00041   int              m, /* LPC order/2                              */
00042   char         *stack
00043 )

00045     spx_word32_t sum;

00049       "P0 = %2;\n\t"           /* P0: coef[m], coef[m-1],..., coef[0] */

00051       "R2 = %1;\n\t"           /* R2: x  */

00053       "R5 = -16383;\n\t"

00060       "R5 = R5.L * R2.L (IS);\n\t"
00061       "R5 = R5 + R4;\n\t"
00062       "R5 >>>= 14;\n\t"
00063       "R3 = R3 + R5;\n\t" 
00064       
00065       "R0 = R2;\n\t"           /* R0: b0 */
00066       "R1 = 16384;\n\t"        /* R1: b1 */

00068       "LOOP_BEGIN cpe%=;\n\t"
