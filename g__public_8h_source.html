00056 typedef struct {
00057         entityState_t   unused;                 // apparently this field was put here accidentally
00058                                                                         //  (and is kept only for compatibility, as a struct pad)
00059 
00060         qboolean        linked;                         // qfalse if not in any good cluster
00061         int                     linkcount;
00062 
00063         int                     svFlags;                        // SVF_NOCLIENT, SVF_BROADCAST, etc
00064 
00065         // only send to this client when SVF_SINGLECLIENT is set        
00066         // if SVF_CLIENTMASK is set, use bitmask for clients to send to (maxclients must be <= 32, up to the mod to enforce this)
00067         int                     singleClient;           
00068 
00069         qboolean        bmodel;                         // if false, assume an explicit mins / maxs bounding box
00070                                                                         // only set by trap_SetBrushModel
00071         vec3_t          mins, maxs;
00072         int                     contents;                       // CONTENTS_TRIGGER, CONTENTS_SOLID, CONTENTS_BODY, etc
00073                                                                         // a non-solid entity should set to 0
00074 
00075         vec3_t          absmin, absmax;         // derived from mins/maxs and origin + rotation
00076 
00077         // currentOrigin will be used for all collision detection and world linking.
00078         // it will not necessarily be the same as the trajectory evaluation for the current
00079         // time, because each entity must be moved one at a time after time is advanced
00080         // to avoid simultanious collision issues
00081         vec3_t          currentOrigin;
00082         vec3_t          currentAngles;
00083 
00084         // when a trace call is made and passEntityNum != ENTITYNUM_NONE,
00085         // an ent will be excluded from testing if:
00086         // ent->s.number == passEntityNum       (don't interact with self)
00087         // ent->r.ownerNum == passEntityNum     (don't interact with your own missiles)
00088         // entity[ent->r.ownerNum].r.ownerNum == passEntityNum  (don't interact with other missiles from owner)
00089         int                     ownerNum;
00090 } entityShared_t;
